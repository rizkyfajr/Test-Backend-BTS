# Test-Backend-BTS

# BTS TEST USING EXPRESSJS
Express.js.

# STACKS

- Node.js 
- Express.js 
- MySQL 
- Sequelize

# HOW TO Install
Git Clone 

# Install Library
Npm install

# CREATE env and edit
 ```bash
      MYSQL_DB_HOST="localhost"
      MYSQL_DB_PORT="3307"
      MYSQL_DB_USER="root"
      MYSQL_DB_PASS="root"
      MYSQL_DB_NAME="bts-rizky"

      JWT_SECRET="secretKey"
      JWT_EXPIRES_IN="8h"
 ```

# MIGRATE
npx sequelize db:migrate


# RUNNING
npm run start
